tekst <- "Prztewarzanie danych w R"
length( tekst )
nchar( tekst )


charToRaw( tekst )
as.integer( charToRaw( tekst ) )


toupper(tekst)

for(i in 1:10){
  cat(sprintf("To jest liczba %d" , i), "\n")
  assign(x = sprintf("x_%d", i), value = 100+i)
}


substr() - wycinanie określonego odcinka tekstu
substr(x= tekst, start = 1, stop = 13 )


strsplit() - oddzielanie poszczególnych słów w wektorze
strsplit(x = tekst, split = " ")


sub() - podmienianie znaków w wektorze (tylko pierwszy element)
sub( pattern = "w", replacement = "_w_", x = tekst)


gsub() - podmienianie znaków w wektorze (wszystkie elementy)
gsub( pattern = "w", replacement = "_w_", x = tekst)




regexpr() - znalezienie pierwszego wystapienia znaku
regexpr(pattern = " ", text = tekst)


gregexpr() - znalezienie wystapienia wszystkich znakow
gregexpr(pattern = " ", text = tekst)




wyrazenie <- c("a", "ab", "acb", "accb", "acccb", "accccb")




# * - co najmniej 0 znaków
grep(pattern = "ac*b", x = wyrazenie, value = T)


# + - co najmniej 1 znak
grep(pattern = "ac+b", x = wyrazenie, value = T)


# ? - co najwyżej 1 znak
grep(pattern = "ac?b", x = wyrazenie, value = T)


# {n,m} - co najmniej n znaków, co najwyżej m znaków
grep(pattern = "ac{2,3}b", x = wyrazenie, value = T)


# Pozycja w tekście
wyrazenie <- c("abcd", "cdab", "cadb","c abd")


# ^ - poczatek tekstu
grep( pattern = "ab", x = wyrazenie, value = T) # wszystkie wystapienia
grep( pattern = "^ab", x = wyrazenie, value = T) # tylko poczatek tekstu


# $ - koniec tekstu
grep( pattern = "ab$", x = wyrazenie, value = T) # wszystkie wystapienia


# Klasy znaków specjalnych
# [::] - definiowanie klasy


# [:digit:] - cyfry
# [:lower:] - litery małe 
# [:upper:] - litery duże
# [:punct:] - znaki interpunkcyjne
# [:blank:] - znaki białe


# [] - definicja znakow, ktore moga pojawic sie w danym miejscu
# [[:punct:][:digit:]]




# [^] - definicja znakow, ktore NIE moga pojawic sie w danym miejscu
# [^[:punct:][:digit:]]




wyrazenie <- c("^ab","ab","abc","abd","abe","ab 100")


grep( pattern = "ab[cd]", x = wyrazenie, value = T) # wszystkie wystapienia
grep( pattern = "ab[]", x = wyrazenie, value = T) # wszystkie wystapienia
